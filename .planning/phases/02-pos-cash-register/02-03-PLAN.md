    **1. Create `src/contexts/transactions-context.tsx`:**

    This context stores transactions in-memory (initializing from mock data) and allows appending new ones. Later phases (Dashboard, Accounting) will read from this same context.

    ```typescript
    'use client'

    import { createContext, useContext, useState, useEffect, type ReactNode } from 'react'
    import type { Transaction } from '@/lib/types'
    import { getTransactions } from '@/lib/data-access'

    interface TransactionsContextType {
      transactions: Transaction[]
      addTransaction: (txn: Transaction) => void
    }

    const TransactionsContext = createContext<TransactionsContextType | null>(null)

    export function TransactionsProvider({ children }: { children: ReactNode }) {
      const [transactions, setTransactions] = useState<Transaction[]>([])

      useEffect(() => {
        getTransactions().then(setTransactions)
      }, [])

      function addTransaction(txn: Transaction) {
        setTransactions(prev => [txn, ...prev])
      }

      return (
        <TransactionsContext value={{ transactions, addTransaction }}>
          {children}
        </TransactionsContext>
      )
    }

    export function useTransactions() {
      const ctx = useContext(TransactionsContext)
      if (!ctx) throw new Error('useTransactions must be used within TransactionsProvider')
      return ctx
    }
    ```

    IMPORTANT: Use React 19 Context syntax: `<TransactionsContext value={...}>` (not `<TransactionsContext.Provider value={...}>`). React 19 supports rendering context directly as a provider.

    Check if this pattern works with Next.js 16. If `getTransactions()` fails because it imports JSON server-side and we're calling from client, the workaround is: initialize with an empty array and load data in useEffect -- the async function in data-access.ts should still work client-side because JSON imports are bundled by webpack/turbopack. If it does NOT work client-side, change the approach: pass initial transactions as a prop from the Server Component level (page.tsx or layout.tsx). This fallback approach would require page.tsx to also fetch transactions and pass them through.

      ```typescript
      function handleCheckout(methode: 'especes' | 'virement') {
        const isResident = state.isBungalowResident
        const total = state.items
          .filter(i => !(isResident && i.type === 'gym-pass'))
          .reduce((sum, i) => sum + i.prixUnitaire * i.quantite, 0)

        const hasGymPass = state.items.some(i => i.type === 'gym-pass')

        const transaction: Transaction = {
          id: `txn-${String(Date.now()).slice(-3)}`,
          date: new Date().toISOString().slice(0, 19),
          type: hasGymPass ? 'gym-pass' : 'fnb',
          centreRevenu: hasGymPass ? 'Gym' : 'F&B',
          clientId: state.client?.id,
          items: state.items.map(item => ({
            produitId: item.produitId,
            nom: item.nom,
            quantite: item.quantite,
            prixUnitaire: item.prixUnitaire,
            sousTotal: item.quantite * item.prixUnitaire,
          })),
          total,
          methode,
        }

        addTransaction(transaction)
        dispatch({ type: 'CLEAR_CART' })
        toast.success('Transaction enregistree', {
          description: `Total: ${total.toLocaleString()} THB`,
        })
      }
      ```
