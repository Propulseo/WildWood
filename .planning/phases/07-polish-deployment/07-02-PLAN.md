    **ResetButton component (defined in the same layout file):**
    ```tsx
    function ResetButton({ isCollapsed }: { isCollapsed: boolean }) {
      const { resetTransactions } = useTransactions()
      const { resetExpenses } = useExpenses()

      function handleReset() {
        resetTransactions()
        resetExpenses()
        toast.success('Donnees reinitialisees')
      }

      return (
        <button
          onClick={handleReset}
          title="Reinitialiser les donnees"
          className={`flex items-center ${

          {!isCollapsed && <span>Reinitialiser</span>}
        </button>
    ```
